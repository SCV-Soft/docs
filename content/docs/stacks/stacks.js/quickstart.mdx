<div className='steps space-y-10'>
  <Step>
    <div className="flex flex-col space-y-3">
      <div className="w-full">
        <p className='text-lg font-bold m-0 dark:text-primary-foreground'>Install Stacks.js</p>
        <div className='flex flex-col space-y-0'>
          <p>First, install the necessary Stacks.js packages in your project:</p>
        </div>
      </div>
      <div className="w-full">
        ```bash title="Terminal"
        npm install @stacks/connect @stacks/transactions
        ```
      </div>
    </div>
  </Step>

  <Step>
    <div className="flex flex-col space-y-3">
      <div className="w-full">
        <p className='text-lg font-bold m-0 dark:text-primary-foreground'>Connect to the Stacks Blockchain</p>
        <div className='flex flex-col space-y-0'>
          <p>Set up the connection to the Stacks blockchain using the `connect` package:</p>
        </div>
      </div>
      <div className="w-full">
        ```tsx
        import { AppConfig, UserSession, showConnect } from '@stacks/connect';

        const appConfig = new AppConfig(['store_write', 'publish_data']);
        const userSession = new UserSession({ appConfig });

        showConnect({
          appDetails: {
            name: 'My App',
            icon: window.location.origin + '/my-app-logo.svg',
          },
          userSession
        });
        ```
      </div>
    </div>
  </Step>

  <Step>
    <div className="flex flex-col space-y-3">
      <div className="w-full">
        <p className='text-lg font-bold m-0 dark:text-primary-foreground'>Make a Contract Call</p>
        <div className='flex flex-col space-y-0'>
          <p>Make a contract call using the `transactions` package:</p>
        </div>
      </div>
      <div className="w-full">
        ```tsx
        import { openContractCall } from '@stacks/connect';
        import { makeContractCall, StacksTestnet } from '@stacks/transactions';

        const network = new StacksTestnet();
        const options = {
          contractAddress: 'ST1234567890abcdef',
          contractName: 'test-contract',
          functionName: 'test-function',
          functionArgs: [],
          network,
          appDetails: {
            name: 'My App',
            icon: window.location.origin + '/my-app-logo.svg',
          },
          onFinish: (data) => {
            console.log('Transaction ID:', data.txId);
            console.log('Raw transaction:', data.txRaw);
          },
        };

        openContractCall(options);
        ```
      </div>
    </div>
  </Step>
</div>
